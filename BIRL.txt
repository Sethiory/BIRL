OH O HOME AI PO (MONSTRO SOMAR(MONSTRO A, MONSTRO B))
      BORA CUMPADE A + B;
BIRL

OH O HOME AI PO (MONSTRO SUBTRAIR(MONSTRO A, MONSTRO B))
      BORA CUMPADE A - B;
BIRL


HORA DO SHOW
      MONSTRO A, B, RES;
      CE QUER VER ESSA PORRA? ("Entra com a e b ai cumpade!!\n");
      QUE QUE CE QUER MONSTRAO? ("%d %d", &A, &B);
      RES = AJUDA O MALUCO TA DOENTE SOMAR(A, B);
      CE QUER VER ESSA PORRA? ("A= %d B= %d\n",A,B);
      CE QUER VER ESSA PORRA? ("SOMA= %d ",RES);
      RES = AJUDA O MALUCO TA DOENTE SUBTRAIR(A, B);
      CE QUER VER ESSA PORRA? ("DIFERENCA= %d ",RES);
      BORA CUMPADE 0;
BIRL
